<script>
async function loadGenesMap() {
    try {
        const r = await fetch('/blogger/genes.txt');
        const t = await r.text();
        const map = {};
        t.split('\n').forEach(l=>{
            const [k,v] = l.split(':').map(s=>s.trim());
            if(k && v) map[k] = v;
        });
        return map;
    } catch(e) { return {}; }
}

function updateStatsDisplay(mutant, level) {
    const stats = window.BloggerUtils.calculateStats(mutant, level);
    if (!stats) return;
    document.getElementById('stat-life').textContent = stats.lifeF;
    document.getElementById('stat-speed').textContent = stats.speedF;
    document.getElementById('meta-speed').textContent = stats.speedF;
    document.getElementById('meta-ability').textContent = stats.ability1Name || '';
    document.getElementById('meta-ability2').textContent = stats.ability2Name || '';
}

async function applyMutantData() {
    const card = document.getElementById('mutant-card');
    if (!card) return;
    const specimen = card.getAttribute('data-specimen');
    if (!specimen) return;
    const mutant = await window.BloggerUtils.getMutantBySpecimen(specimen);
    if (!mutant) return;
    // fill basic fields
    document.getElementById('mutant-name').textContent = mutant.name;
    document.getElementById('mutant-story').textContent = mutant.description || '';
    // images (optional adjust path depending on naming)
    const thumb = document.querySelector('.mutant-image');
    if (thumb) thumb.src = `https://s-ak.kobojo.com/mutants/assets/thumbnails/${mutant.specimen.toLowerCase()}.png`;
    // set genes icons
    const genesMap = await loadGenesMap();
    const dna = mutant.dna || '';
    if (dna.length > 0 && document.getElementById('gene-1'))
        document.getElementById('gene-1').src = genesMap[dna[0]] || '';
    if (dna.length > 1 && document.getElementById('gene-2'))
        document.getElementById('gene-2').src = genesMap[dna[1]] || '';
    // initial stats at default level
    const lvl = parseInt(document.getElementById('fame-level-input').value) || 25;
    updateStatsDisplay(mutant, lvl);
    document.getElementById('update-stats-btn').addEventListener('click', () => {
        const newLvl = parseInt(document.getElementById('fame-level-input').value) || 25;
        updateStatsDisplay(mutant, newLvl);
    });
}

document.addEventListener('DOMContentLoaded', applyMutantData);
</script>
